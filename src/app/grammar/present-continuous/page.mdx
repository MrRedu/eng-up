<GrammarExample
  structure="Subject + to be (am/is/are) + Verb-ing + Complement"
  examples={[

      translation: 'Estoy estudiando inglés ahora.',
    },

      text: 'He is studying English now.',
      translation: 'Él está estudiando inglés ahora.',
    },
    {
      type: 'Negative',
      text: "They are not (aren't) studying English.",
      translation: 'Ellos no están estudiando inglés.',
    },
    {

      text: 'Are you studying English?',
      translation: '¿Estás estudiando inglés?',
    },
    { type: 'Answer (+)', text: 'Yes, I am.', translation: 'Sí, lo estoy.' },
    {
      type: 'Answer (-)',
      text: 'No, I am not.',
      translation: 'No, no lo estoy.',
    },
  ]}
/>

<Alert variant="warn" title="Recordatorio">
  El verbo **'to be'** cambia según el sujeto: **I am, You are, He/She/It is**.
</Alert>
