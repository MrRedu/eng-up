# Present Perfect

<Typography variant="p" className="text-center">
  {`Se utiliza para acciones que ocurrieron en un tiempo no específico del pasado o que tienen relevancia en el presente.`}
</Typography>

<br />
<br />

<GrammarExample
  structure="Subject + have/has + Past Participle + Complement"
  examples={[
    {
      type: '1st Person (Affirmative)',
      text: 'I have studied all the topics.',
      translation: 'He estudiado todos los temas.',
    },
    {
      type: '3rd Person (Affirmative)',
      text: 'She has studied all the topics.',
      translation: 'Ella ha estudiado todos los temas.',
    },
    {
      type: 'Negative',
      text: "We have not (haven't) finished yet.",
      translation: 'No hemos terminado todavía.',
    },
    {
      type: 'Interrogative',
      text: 'Have you seen my notebook?',
      translation: '¿Has visto mi cuaderno?',
    },
    {
      type: 'Answer (+)',
      text: 'Yes, I have.',
      translation: 'Sí, lo he hecho.',
    },
    {
      type: 'Answer (-)',
      text: "No, I haven't.",
      translation: 'No, no lo he hecho.',
    },
  ]}
/>

<Alert variant="warn" title="Recordatorio">
  Usamos **'has'** para **He, She, It** y **'have'** para el resto.
</Alert>
