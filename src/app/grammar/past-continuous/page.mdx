# Past Continuous

<Typography variant="p" className="text-center">
  {`Se utiliza para describir acciones que estaban en progreso en un momento específico del pasado.`}
</Typography>

<br />
<br />

<GrammarExample
  structure="Subject + to be (was/were) + Verb-ing + Complement"
  examples={[
    {
      type: '1st Person (Affirmative)',
      text: 'I was studying when you called.',
      translation: 'Estaba estudiando cuando llamaste.',
    },
    {
      type: 'Plural (Affirmative)',
      text: 'They were studying all night.',
      translation: 'Ellos estuvieron estudiando toda la noche.',
    },
    {
      type: 'Negative',
      text: "He was not (wasn't) studying.",
      translation: 'Él no estaba estudiando.',
    },
    {
      type: 'Interrogative',
      text: 'Were you studying?',
      translation: '¿Estabas estudiando?',
    },
    { type: 'Answer (+)', text: 'Yes, I was.', translation: 'Sí, lo estaba.' },
    {
      type: 'Answer (-)',
      text: "No, I wasn't.",
      translation: 'No, no lo estaba.',
    },
  ]}
/>

<Alert variant="info" title="Dato importante">
  Usamos **'was'** para I, He, She, It y **'were'** para You, We, They. 
</Alert>
